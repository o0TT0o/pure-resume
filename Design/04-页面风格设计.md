# 页面风格设计方案

## 设计理念
- **简洁至上**：去除多余装饰，突出核心内容
- **清晰易读**：合理的排版和间距
- **专业美观**：参考BOSS直聘等主流招聘网站
- **响应式设计**：适配各种设备
- **多模板支持**：提供多种风格模板，满足不同场景需求（配置驱动）
- **多语言界面**：支持多语言界面，适配不同地区用户（配置驱动）
- **Tailwind CSS**：2026年主流CSS框架，快速迭代样式
- **配置驱动样式**：样式根据配置文件动态生成，消除硬编码
- ⚠️ **重要**：语言切换仅加载用户预设的简历内容，不提供翻译功能
- ⚠️ **重要**：不使用暗黑模式，保持简洁的浅色主题

**配置驱动的渲染**：
```javascript
// 根据 data/skills/config.json 动态渲染技能分类
async function renderSkills() {
  const skillsConfig = await api.getSkillCategories();
  const skillsData = await api.getData('skills');

  return skillsConfig.categories.map(category => {
    const categorySkills = skillsData[category.id] || [];
    let skillsHTML = '';

    // 根据配置的类型渲染不同的样式
    if (category.type === 'rated') {
      skillsHTML = categorySkills.map(skill => `
        <div class="skill-item">
          <span>${skill.name}</span>
          <div class="skill-bar">
            <div class="skill-progress" style="width: ${skill.level / category.ratingScale * 100}%"></div>
          </div>
        </div>
      `).join('');
    } else if (category.type === 'text') {
      skillsHTML = categorySkills.map(skill => `
        <div class="skill-item">
          <span>${skill.language}</span>
          <span class="skill-level">${skill.level}</span>
        </div>
      `).join('');
    } else if (category.type === 'list') {
      skillsHTML = categorySkills.map(skill => `
        <div class="skill-item">
          <span>${skill}</span>
        </div>
      `).join('');
    }

    return `
      <div class="skill-category">
        <h3 class="skill-category-title">
          <span class="skill-category-icon">${category.icon}</span>
          ${category.name}
        </h3>
        <div class="skill-${category.type}">
          ${skillsHTML}
        </div>
      </div>
    `;
  }).join('');
}
```

### 编辑器集成
在编辑器中，用户修改数据后点击保存按钮时触发版本对比：

```javascript
// 编辑器保存流程
async function handleSave(langCode, module, newData) {
  // 1. 保存当前版本快照
  await versionManager.saveSnapshot(langCode, module);

  // 2. 对比版本差异
  const comparison = await versionManager.compareChanges(langCode, module, newData);

  // 3. 如果有变化，显示对比对话框
  if (comparison.hasChanges) {
    versionManager.showComparison(comparison);
  } else {
    // 没有变化，直接提示
    alert('没有检测到任何修改');
  }
}

// 用户确认保存
async function confirmSave() {
  // 保存数据到服务器
  await api.saveResumeData(languageManager.currentLang, currentModule, editedData);

  // 关闭模态框
  closeModal();

  // 显示成功提示
  showToast('保存成功！');
}
```

## 模板切换实现

### JavaScript 模板管理
```javascript
class TemplateRenderer {
  constructor() {
    this.currentTemplate = localStorage.getItem('template') || 'boss-zhipin';
    this.body = document.body;
  }

  async switchTemplate(templateId) {
    // 移除旧模板类
    this.body.classList.remove(`template-${this.currentTemplate}`);

    // 加载新模板样式
    const templateConfig = await templateManager.getTemplate(templateId);
    const styleElement = document.getElementById('dynamic-template-style');
    styleElement.textContent = templateConfig.style;

    // 应用新模板类
    this.currentTemplate = templateId;
    this.body.classList.add(`template-${templateId}`);
    localStorage.setItem('template', templateId);

    // 重新渲染内容（使用当前语言的数据）
    await this.render();
  }

  async render() {
    // 使用语言管理器获取当前语言的简历数据
    const data = {
      personal: languageManager.getData('personal'),
      experience: languageManager.getData('experience'),
      projects: languageManager.getData('projects'),
      education: languageManager.getData('education'),
      certificates: languageManager.getData('certificates'),
      skills: languageManager.getData('skills')
    };

    const html = await templateManager.renderTemplate(this.currentTemplate, data);
    document.querySelector('.resume-content').innerHTML = html;

    // 检查数据是否为空，显示提示
    this.checkDataStatus(data);
  }

  checkDataStatus(data) {
    const isEmpty = Object.values(data).every(module =>
      Object.keys(module).length === 0
    );

    const notice = document.getElementById('languageNotice');
    notice.style.display = isEmpty ? 'block' : 'none';
  }
}
```
