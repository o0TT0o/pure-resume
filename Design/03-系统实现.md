# 系统实现方案

## 开发原则

### 代码质量控制
- **文件大小控制**: 每个功能实现文件必须控制在合理大小范围内，避免单文件过于臃肿
- **代码行数管理**: 控制每个文件的代码行数，建议单个文件不超过 500-800 行（核心模块除外）
- **代码去重**: 严格避免重复代码和重复定义，通过函数封装、模块化、继承等方式消除重复
- **可读性维护**: 保持代码清晰可读，包括合理的命名、适当的注释、一致的代码风格

### 状态管理
- **用户设置持久化**: 记住用户的基础设置（如语言偏好、模板选择、编辑器配置等）
- **状态保持**: 在页面切换、刷新时不能丢失用户的基础设置
- **本地存储**: 使用 localStorage 或 sessionStorage 存储用户偏好设置
- **服务端同步**: 对于需要持久化的用户设置，通过 API 保存到数据文件
- **状态同步**: 注意数据产生变化后的状态同步，包括：
  - 编辑操作后立即更新UI状态
  - 数据变化时同步更新保存按钮状态（有变化时启用，无变化时禁用）
  - 保存成功后重置原始数据状态，确保状态一致性
  - 重置操作后同步更新相关UI状态

### 模块一致性
- **统一模块标准**: 每个功能模块必须保持一致性，包括接口设计、数据结构、UI 风格
- **模块化实现**: 通过模块化和模板化实现功能，避免模块间有细节出入
- **共享组件**: 抽取公共组件和工具函数，确保不同模块使用相同的实现
- **配置驱动**: 使用配置文件定义模块行为，确保所有模块遵循统一的规范

```javascript
const express = require('express');
const cors = require('cors');
const helmet = require('helmet');
const compression = require('compression');
const path = require('path');

const dataRoutes = require('./routes/data');
const uploadRoutes = require('./routes/upload');
const exportRoutes = require('./routes/export');
const importRoutes = require('./routes/import');
const backupRoutes = require('./routes/backup');
const languageRoutes = require('./routes/language');
const templateRoutes = require('./routes/template');
const versionRoutes = require('./routes/version');
const configRoutes = require('./routes/config'); // 新增：配置路由

  });
  next();

// 路由

  res.json({
    status: 'ok',
    timestamp: new Date().toISOString(),
    uptime: process.uptime()
  });
});

// 404处理
app.use((req, res) => {
  res.status(404).json({
    error: {
      code: 'NOT_FOUND',
      message: '请求的资源不存在',

      timestamp: new Date().toISOString()
    }

    environment: process.env.NODE_ENV || 'development',

  process.exit(0);

  process.exit(0);

  process.exit(1);
});

process.on('unhandledRejection', (reason, promise) => {
```

### 2. 数据服务 (dataService.js)
```javascript
const fs = require('fs').promises;
const path = require('path');
const { validateData } = require('../utils/validateHelper');
const { backupFile } = require('./backupService');

const DATA_DIR = path.join(__dirname, '../data');

class DataService {
  async read(filename) {
    const filePath = path.join(DATA_DIR, `${filename}.json`);
    const data = await fs.readFile(filePath, 'utf8');
    return JSON.parse(data);
  }

  async write(filename, data) {
    // 验证数据
    const isValid = await validateData(filename, data);
    if (!isValid) {
      throw new Error('数据验证失败');
    }

    // 备份原文件
    await backupFile(filename);

    // 写入新数据
    const filePath = path.join(DATA_DIR, `${filename}.json`);
    await fs.writeFile(filePath, JSON.stringify(data, null, 2), 'utf8');
  }
}

module.exports = new DataService();
```

### 4. 语言服务 (languageService.js)
```javascript
const fs = require('fs').promises;
const path = require('path');

const LANGUAGES_DIR = path.join(__dirname, '../languages');
const DATA_DIR = path.join(__dirname, '../data');

class LanguageService {
  // 获取支持的语言列表
  async getLanguageList() {
    const indexPath = path.join(LANGUAGES_DIR, 'index.json');
    const data = await fs.readFile(indexPath, 'utf8');
    return JSON.parse(data);
  }

  // 获取系统界面语言包
  async getLanguagePack(langCode) {
    const langPath = path.join(LANGUAGES_DIR, `${langCode}.json`);
    const data = await fs.readFile(langPath, 'utf8');
    return JSON.parse(data);
  }

  // 获取指定语言的简历数据
  async getResumeData(langCode, module) {
    const langDir = path.join(DATA_DIR, langCode);
    const filePath = path.join(langDir, `${module}.json`);

    // 如果文件不存在，返回空对象
    try {
      const data = await fs.readFile(filePath, 'utf8');
      return JSON.parse(data);
    } catch (error) {
      if (error.code === 'ENOENT') {
        return {}; // 语言数据不存在，返回空对象
      }
      throw error;
    }
  }

  // 保存指定语言的简历数据
  async saveResumeData(langCode, module, data) {
    const langDir = path.join(DATA_DIR, langCode);

    // 确保语言目录存在
    await fs.mkdir(langDir, { recursive: true });

    // 备份原文件
    const filePath = path.join(langDir, `${module}.json`);
    try {
      await fs.readFile(filePath); // 检查文件是否存在
      await this.backupFile(langCode, module);
    } catch (error) {
      // 文件不存在，无需备份
    }

    // 写入新数据
    await fs.writeFile(filePath, JSON.stringify(data, null, 2), 'utf8');
  }

  // 备份语言数据文件
  async backupFile(langCode, module) {
    const sourcePath = path.join(DATA_DIR, langCode, `${module}.json`);
    const backupPath = path.join(DATA_DIR, langCode, `.${module}.${Date.now()}.json`);
    await fs.copyFile(sourcePath, backupPath);
  }

  // 复制数据到另一种语言
  async copyData(sourceLang, targetLang, module) {
    const sourceData = await this.getResumeData(sourceLang, module);
    await this.saveResumeData(targetLang, module, sourceData);
  }

  // 获取语言数据状态（是否为空）
  async getLanguageStatus(langCode) {
    const modules = ['personal', 'experience', 'projects', 'education', 'certificates', 'skills'];
    const status = {};

    for (const module of modules) {
      const data = await this.getResumeData(langCode, module);
      const isEmpty = Object.keys(data).length === 0;
      status[module] = { exists: !isEmpty, data };
    }

    return status;
  }
}

module.exports = new LanguageService();
```

### 5. 模板服务 (templateService.js) - Tailwind CSS版本（支持国际化）
```javascript
const fs = require('fs').promises;
const path = require('path');
const handlebars = require('handlebars');
const marked = require('marked'); // Markdown解析

const TEMPLATES_DIR = path.join(__dirname, '../templates');
const LANGUAGES_DIR = path.join(__dirname, '../languages');

class TemplateService {
  async getTemplateList() {
    const templateDirs = await fs.readdir(TEMPLATES_DIR);
    const templates = [];

    for (const dir of templateDirs) {
      const configPath = path.join(TEMPLATES_DIR, dir, 'config.json');
      const config = JSON.parse(await fs.readFile(configPath, 'utf8'));
      templates.push(config);
    }

    return templates;
  }

  async getTemplate(templateId) {
    const configPath = path.join(TEMPLATES_DIR, templateId, 'config.json');
    const contentPath = path.join(TEMPLATES_DIR, templateId, 'content.md');
    const templatePath = path.join(TEMPLATES_DIR, templateId, 'template.html');
    const tailwindConfigPath = path.join(TEMPLATES_DIR, templateId, 'tailwind.config.js');

    const config = JSON.parse(await fs.readFile(configPath, 'utf8'));
    const content = await fs.readFile(contentPath, 'utf8');
    const template = await fs.readFile(templatePath, 'utf8');

    // 读取Tailwind配置（如果存在）
    let tailwindConfig = null;
    try {
      tailwindConfig = await fs.readFile(tailwindConfigPath, 'utf8');
    } catch (error) {
      // 使用默认配置
    }

    return { config, content, template, tailwindConfig };
  }

  // 获取界面语言包
  async getLanguagePack(langCode) {
    const langPath = path.join(LANGUAGES_DIR, `${langCode}.json`);
    try {
      const data = await fs.readFile(langPath, 'utf8');
      return JSON.parse(data);
    } catch (error) {
      // 语言包不存在，返回空对象
      return {};
    }
  }

  // 预处理 Markdown 内容中的国际化标记
  preprocessContent(content, languagePack) {
    // 替换 {{i18n.xxx}} 为实际文本
    let processedContent = content;
    
    // 正则匹配 {{i18n.xxx.yyy}} 格式的变量
    const i18nRegex = /\{\{i18n\.([a-zA-Z0-9_.]+)\}\}/g;
    
    processedContent = processedContent.replace(i18nRegex, (match, keyPath) => {
      const keys = keyPath.split('.');
      let value = languagePack;
      
      for (const key of keys) {
        value = value?.[key];
      }
      
      return value || match; // 如果找不到翻译，保留原标记
    });

    return processedContent;
  }

  async renderTemplate(templateId, data, langCode = 'zh-CN') {
    const { template, content } = await this.getTemplate(templateId);

    // 获取界面语言包
    const languagePack = await this.getLanguagePack(langCode);

    // 预处理 Markdown 内容，替换国际化标记
    const preprocessedContent = this.preprocessContent(content, languagePack);

    // 解析Markdown内容
    const parsedContent = marked(preprocessedContent);

    // 使用Handlebars组合模板和数据
    const compiledTemplate = handlebars.compile(template);
    
    return compiledTemplate({
      ...data,                  // 简历数据（来自 data/[lang]/）
      i18n: languagePack,       // 界面语言包（来自 languages/[lang].json）
      langCode: langCode,        // 当前语言代码
      content: parsedContent     // 处理后的 Markdown 内容
    });
  }

  async applyTemplate(templateId, data, langCode = 'zh-CN') {
    const { config, tailwindConfig } = await this.getTemplate(templateId);
    const html = await this.renderTemplate(templateId, data, langCode);
    return { html, config, tailwindConfig };
  }
}

module.exports = new TemplateService();
```

### 6. 版本服务 (versionService.js)
```javascript
const fs = require('fs').promises;
const path = require('path');
const diff = require('diff'); // 用于版本对比

const DATA_DIR = path.join(__dirname, '../data');

class VersionService {
  // 保存前记录当前版本（用于对比）
  async saveCurrentVersion(langCode, module) {
    const langDir = path.join(DATA_DIR, langCode);
    const filePath = path.join(langDir, `${module}.json`);

    try {
      const currentData = await fs.readFile(filePath, 'utf8');
      const versionsPath = path.join(langDir, `.${module}.versions.json`);

      let versions = [];
      try {
        versions = JSON.parse(await fs.readFile(versionsPath, 'utf8'));
      } catch (error) {
        // 文件不存在，使用空数组
      }

      // 保存当前版本快照
      versions.push({
        id: `v${Date.now()}`,
        timestamp: new Date().toISOString(),
        data: JSON.parse(currentData)
      });

      // 只保留最近10个版本
      if (versions.length > 10) {
        versions = versions.slice(-10);
      }

      await fs.writeFile(versionsPath, JSON.stringify(versions, null, 2), 'utf8');
    } catch (error) {
      // 文件不存在，无需记录
    }
  }

  // 获取最新版本快照
  async getLatestVersion(langCode, module) {
    const versionsPath = path.join(DATA_DIR, langCode, `.${module}.versions.json`);

    try {
      const versions = JSON.parse(await fs.readFile(versionsPath, 'utf8'));
      return versions[versions.length - 1] || null;
    } catch (error) {
      return null;
    }
  }

  // 对比当前版本和修改后的数据
  async compareVersions(langCode, module, newData) {
    const latestVersion = await this.getLatestVersion(langCode, module);

    if (!latestVersion) {
      return {
        hasChanges: Object.keys(newData).length > 0,
        changes: []
      };
    }

    const oldData = latestVersion.data;
    const changes = [];

    // 深度对比对象
    this.deepCompare(oldData, newData, '', changes);

    return {
      hasChanges: changes.length > 0,
      changes: changes,
      oldVersion: latestVersion,
      newVersion: {
        id: `v${Date.now()}`,
        timestamp: new Date().toISOString(),
        data: newData
      }
    };
  }

  // 深度对比辅助方法
  deepCompare(oldObj, newObj, path, changes) {
    // 如果是数组
    if (Array.isArray(oldObj) && Array.isArray(newObj)) {
      // 对比数组长度
      if (oldObj.length !== newObj.length) {
        changes.push({
          type: 'array_length',
          path: path,
          old: oldObj.length,
          new: newObj.length
        });
      }

      // 对比数组元素
      const maxLength = Math.max(oldObj.length, newObj.length);
      for (let i = 0; i < maxLength; i++) {
        this.deepCompare(
          oldObj[i],
          newObj[i],
          `${path}[${i}]`,
          changes
        );
      }
      return;
    }

    // 如果是对象
    if (typeof oldObj === 'object' && oldObj !== null &&
        typeof newObj === 'object' && newObj !== null) {
      const keys = new Set([
        ...Object.keys(oldObj),
        ...Object.keys(newObj)
      ]);

      keys.forEach(key => {
        const oldVal = oldObj[key];
        const newVal = newObj[key];

        // 检查新增字段
        if (!(key in oldObj) && key in newObj) {
          changes.push({
            type: 'added',
            path: `${path}.${key}`,
            new: newVal
          });
        }
        // 检查删除字段
        else if (key in oldObj && !(key in newObj)) {
          changes.push({
            type: 'removed',
            path: `${path}.${key}`,
            old: oldVal
          });
        }
        // 递归对比
        else if (typeof oldVal === 'object' && typeof newVal === 'object') {
          this.deepCompare(oldVal, newVal, `${path}.${key}`, changes);
        }
        // 值对比
        else if (JSON.stringify(oldVal) !== JSON.stringify(newVal)) {
          changes.push({
            type: 'modified',
            path: `${path}.${key}`,
            old: oldVal,
            new: newVal
          });
        }
      });
      return;
    }

    // 基本类型对比
    if (JSON.stringify(oldObj) !== JSON.stringify(newObj)) {
      changes.push({
        type: 'modified',
        path: path,
        old: oldObj,
        new: newObj
      });
    }
  }

  // 清理旧版本（保留最近10个）
  async cleanupOldVersions(langCode, module, keepCount = 10) {
    const versionsPath = path.join(DATA_DIR, langCode, `.${module}.versions.json`);

    try {
      const versions = JSON.parse(await fs.readFile(versionsPath, 'utf8'));

      if (versions.length > keepCount) {
        const trimmedVersions = versions.slice(-keepCount);
        await fs.writeFile(
          versionsPath,
          JSON.stringify(trimmedVersions, null, 2),
          'utf8'
        );
      }
    } catch (error) {
      // 文件不存在或出错，忽略
    }
  }
}

module.exports = new VersionService();
```

```javascript
const Ajv = require('ajv');
const fs = require('fs').promises;
const path = require('path');

class ConfigValidator {
  constructor() {
    this.ajv = new Ajv({ allErrors: true });
    this.schemas = {};
  }

  // 加载所有 Schema
  async loadSchemas() {
    const schemaDir = path.join(__dirname, '../schemas');
    try {
      const schemaFiles = await fs.readdir(schemaDir);

      for (const file of schemaFiles) {
        if (file.endsWith('.schema.json')) {
          const schemaPath = path.join(schemaDir, file);
          const schemaContent = await fs.readFile(schemaPath, 'utf8');
          const schema = JSON.parse(schemaContent);

          // 添加到 Ajv
          const schemaName = file.replace('.schema.json', '');
          this.schemas[schemaName] = this.ajv.compile(schema);
          logger.info(`Schema loaded: ${schemaName}`);
        }
      }
    } catch (error) {
      logger.error('Failed to load schemas:', error);
      throw new Error(`配置Schema加载失败: ${error.message}`);
    }
  }

  // 验证配置文件
  async validateConfig(configPath, schemaName) {
    const schema = this.schemas[schemaName];
    if (!schema) {
      throw new Error(`Schema not found: ${schemaName}`);
    }

    try {
      const configContent = await fs.readFile(configPath, 'utf8');
      const config = JSON.parse(configContent);

      const valid = schema(config);

      if (!valid) {
        const errors = schema.errors.map(err => ({
          path: err.instancePath,
          message: err.message,
          params: err.params
        }));

        logger.warn('Config validation failed', {
          configPath,
          schemaName,
          errors
        });

        throw new ConfigValidationError(
          `配置验证失败: ${configPath}`,
          errors
        );
      }

      logger.info('Config validation passed', { configPath, schemaName });
      return true;
    } catch (error) {
      if (error instanceof ConfigValidationError) {
        throw error;
      }
      logger.error('Config validation error:', error);
      throw new Error(`配置文件读取或解析失败: ${error.message}`);
    }
  }

  // 批量验证所有配置
  async validateAllConfigs() {
    const results = [];

    // 验证模块配置
    try {
      await this.validateConfig(
        path.join(__dirname, '../data/modules/config.json'),
        'modules'
      );
      results.push({ file: 'modules/config.json', status: 'valid' });
    } catch (error) {
      results.push({ file: 'modules/config.json', status: 'invalid', error: error.errors });
    }

    // 验证技能配置
    try {
      await this.validateConfig(
        path.join(__dirname, '../data/skills/config.json'),
        'skills'
      );
      results.push({ file: 'skills/config.json', status: 'valid' });
    } catch (error) {
      results.push({ file: 'skills/config.json', status: 'invalid', error: error.errors });
    }

    // 验证联系字段配置
    try {
      await this.validateConfig(
        path.join(__dirname, '../data/personal/fields-config.json'),
        'contact-fields'
      );
      results.push({ file: 'personal/fields-config.json', status: 'valid' });
    } catch (error) {
      results.push({ file: 'personal/fields-config.json', status: 'invalid', error: error.errors });
    }

    // 验证列表配置
    try {
      await this.validateConfig(
        path.join(__dirname, '../data/lists/config.json'),
        'lists'
      );
      results.push({ file: 'lists/config.json', status: 'valid' });
    } catch (error) {
      results.push({ file: 'lists/config.json', status: 'invalid', error: error.errors });
    }

    return results;
  }
}
```

```javascript
function errorHandler(err, req, res, next) {
  // 记录错误日志
  logger.error('Error occurred:', {
    error: err.message,
    stack: err.stack,
    url: req.url,
    method: req.method,
    body: req.body,
    query: req.query,
    params: req.params,
    ip: req.ip,
    userAgent: req.get('user-agent')
  });

  // 判断是否为自定义错误
  if (err instanceof AppError) {
    return res.status(err.statusCode).json({
      error: {
        code: err.code,
        message: err.message,
        details: err.details,
        timestamp: err.timestamp
      }
    });
  }

  // 配置验证错误特殊处理
  if (err instanceof ConfigValidationError) {
    return res.status(400).json({
      error: {
        code: 'CONFIG_VALIDATION_ERROR',
        message: '配置验证失败',
        details: {
          errors: err.errors.map(e => ({
            path: e.path,
            message: e.message,
            suggestion: getSuggestion(e)
          }))
        },
        timestamp: new Date().toISOString()
      }
    });
  }

  // 未知错误
  logger.error('Unhandled error:', err);
  res.status(500).json({
    error: {
      code: 'INTERNAL_SERVER_ERROR',
      message: '服务器内部错误',
      timestamp: new Date().toISOString()
    }
  });
}

// 获取错误提示建议
function getSuggestion(error) {
  const { path, message, params } = error;

  if (message.includes('required')) {
    return `缺少必填字段: ${path}`;
  }
  if (message.includes('pattern')) {
    return `格式不匹配，请参考配置Schema中的pattern规则`;
  }
  if (message.includes('enum')) {
    return `值不在允许的枚举范围内: ${JSON.stringify(params.allowedValues)}`;
  }
  if (message.includes('minimum') || message.includes('maximum')) {
    return `值范围错误: ${JSON.stringify(params)}`;
  }

  return '请检查配置文件格式和内容';
}
```

```javascript
    await configService.init();

    const hasErrors = validationResults.some(r => r.status === 'invalid');

    }

    logger.info('All configurations validated successfully');

    logger.error('Configuration initialization failed:', error);
    throw error;
  }
```

**语言包结构说明**：
实际实现采用子目录+双文件结构：
- `languages/[langCode]/config-i18n.json`：配置项翻译（模块名、字段名等）
- `languages/[langCode]/ui.json`：系统界面翻译（按钮、标签、提示等）
- 相比设计文档的单文件结构，这种分离方式职责更清晰，更易维护和扩展

```javascript
// 前端语言管理器
class LanguageManager {
  constructor() {
    this.currentLang = localStorage.getItem('language') || 'zh-CN';
    this.uiTranslations = {}; // 系统界面翻译
    this.resumeData = {};     // 简历数据
  }

  async init() {
    await this.loadUILanguage(this.currentLang);
    await this.loadResumeData(this.currentLang);
  }

  // 加载系统界面语言包
  async loadUILanguage(langCode) {
    this.uiTranslations = await api.getLanguagePack(langCode);
  }

  // 加载简历数据
  async loadResumeData(langCode) {
    const modules = ['personal', 'experience', 'projects', 'education', 'certificates', 'skills'];
    this.resumeData[langCode] = {};

    for (const module of modules) {
      try {
        this.resumeData[langCode][module] = await api.getResumeData(langCode, module);
      } catch (error) {
        this.resumeData[langCode][module] = {}; // 数据不存在时使用空对象
      }
    }
  }

  // 切换语言
  async switchLanguage(langCode) {
    // 加载新的界面语言
    await this.loadUILanguage(langCode);

    // 加载新的简历数据
    if (!this.resumeData[langCode]) {
      await this.loadResumeData(langCode);
    }

    // 更新当前语言
    this.currentLang = langCode;
    localStorage.setItem('language', langCode);

    // 刷新页面内容
    this.render();
  }

  // 获取界面翻译文本
  t(key) {
    const keys = key.split('.');
    let value = this.uiTranslations;
    for (const k of keys) {
      value = value?.[k];
      if (value === undefined) return key;
    }
    return value;
  }

  // 获取当前语言的简历数据
  getData(module) {
    return this.resumeData[this.currentLang]?.[module] || {};
  }

  // 保存当前语言的简历数据
  async saveData(module, data) {
    await api.saveResumeData(this.currentLang, module, data);
    this.resumeData[this.currentLang][module] = data;
  }

  // 复制数据到另一种语言
  async copyData(sourceLang, targetLang, module) {
    await api.copyData(sourceLang, targetLang, module);
    if (!this.resumeData[targetLang]) {
      await this.loadResumeData(targetLang);
    }
  }

  // 渲染页面
  render() {
    // 更新所有界面文本
    document.querySelectorAll('[data-i18n]').forEach(el => {
      const key = el.getAttribute('data-i18n');
      el.textContent = this.t(key);
    });

    // 更新简历内容
    this.renderResume();
  }

  // 渲染简历内容
  renderResume() {
    // 根据当前模板渲染简历数据
    const data = {
      personal: this.getData('personal'),
      experience: this.getData('experience'),
      projects: this.getData('projects'),
      education: this.getData('education'),
      certificates: this.getData('certificates'),
      skills: this.getData('skills')
    };

    templateManager.renderResume(data);
  }
}

// 前端版本管理器
class VersionManager {
  constructor() {
    this.currentVersions = {};
  }

  // 编辑前保存快照
  async saveSnapshot(langCode, module) {
    await api.saveSnapshot(langCode, module);
  }

  // 对比修改
  async compareChanges(langCode, module, newData) {
    return await api.compareVersions(langCode, module, newData);
  }

  // 显示对比结果
  showComparison(result) {
    if (!result.hasChanges) {
      alert('没有检测到任何修改');
      return;
    }

    let comparisonHTML = '<div class="version-comparison">';

    result.changes.forEach(change => {
      const icon = this.getChangeIcon(change.type);
      const color = this.getChangeColor(change.type);

      comparisonHTML += `
        <div class="change-item change-${change.type}" style="border-left: 3px solid ${color}">
          <div class="change-header">
            <span class="change-icon">${icon}</span>
            <span class="change-path">${change.path}</span>
            <span class="change-type">${this.getChangeLabel(change.type)}</span>
          </div>
          ${this.renderChangeDetail(change)}
        </div>
      `;
    });

    comparisonHTML += '</div>';

    const modal = document.createElement('div');
    modal.className = 'version-modal';
    modal.innerHTML = `
      <div class="modal-overlay" onclick="this.parentElement.remove()"></div>
      <div class="modal-content">
        <div class="modal-header">
          <h3>版本对比</h3>
          <button class="modal-close" onclick="this.closest('.version-modal').remove()">✕</button>
        </div>
        <div class="modal-body">
          ${comparisonHTML}
        </div>
        <div class="modal-footer">
          <button class="btn-secondary" onclick="this.closest('.version-modal').remove()">取消</button>
          <button class="btn-primary" onclick="this.closest('.version-modal').remove(); saveChanges()">确认保存</button>
        </div>
      </div>
    `;

    document.body.appendChild(modal);
  }

  getChangeIcon(type) {
    const icons = {
      'added': '+',
      'removed': '-',
      'modified': '→',
      'array_length': '📊'
    };
    return icons[type] || '•';
  }

  getChangeColor(type) {
    const colors = {
      'added': '#52C41A',
      'removed': '#FF4D4F',
      'modified': '#FAAD14',
      'array_length': '#2E54FF'
    };
    return colors[type] || '#999999';
  }

  getChangeLabel(type) {
    const labels = {
      'added': '新增',
      'removed': '删除',
      'modified': '修改',
      'array_length': '数组长度变化'
    };
    return labels[type] || type;
  }

  renderChangeDetail(change) {
    if (change.type === 'array_length') {
      return `
        <div class="change-detail">
          <span class="old-value">${change.old} 项</span>
          <span>→</span>
          <span class="new-value">${change.new} 项</span>
        </div>
      `;
    }

    return `
      <div class="change-detail">
        ${change.old !== undefined ? `<div class="old-value">${this.formatValue(change.old)}</div>` : ''}
        <div class="arrow">↓</div>
        ${change.new !== undefined ? `<div class="new-value">${this.formatValue(change.new)}</div>` : ''}
      </div>
    `;
  }

  formatValue(value) {
    if (typeof value === 'object' && value !== null) {
      return JSON.stringify(value, null, 2);
    }
    return String(value);
  }
}
```
