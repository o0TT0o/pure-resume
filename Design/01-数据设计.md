# 数据设计方案

## 核心原则
- **不使用数据库**：所有数据以文件形式存储
- **动态读写**：查看时读取，保存时写入
- **配置驱动**：消除硬编码，通过配置文件动态管理
- **易于扩展**：添加新语言、新模块无需修改代码

## 数据存储方案

- ✅ **配置驱动**：所有可变配置从代码移至配置文件

- ✅ **i18n支持**：配置文本通过 i18nKey 引用，支持多语言
- ✅ **向后兼容**：保留原有数据结构，平滑升级

#### 11. 配置驱动架构 - 消除硬编码与验证机制

##### 11.1 模块配置 (data/modules/config.json)
动态定义数据模块，添加新模块无需修改代码，支持版本管理和Schema验证：

```json
{
  "_meta": {
    "version": "1.0.0",
    "lastModified": "2026-03-18T10:00:00Z",
    "schema": "modules",
    "description": "模块配置文件"
  },
  "personal": {
    "id": "personal",
    "i18nKey": "module.personal",
    "required": true,
    "file": "personal.json",
    "type": "object"
  },
  "experience": {
    "id": "experience",
    "i18nKey": "module.experience",
    "required": true,
    "file": "experience.json",
    "type": "array"
  },
  "projects": {
    "id": "projects",
    "i18nKey": "module.projects",
    "required": false,
    "file": "projects.json",
    "type": "array"
  },
  "education": {
    "id": "education",
    "i18nKey": "module.education",
    "required": false,
    "file": "education.json",
    "type": "array"
  },
  "certificates": {
    "id": "certificates",
    "i18nKey": "module.certificates",
    "required": false,
    "file": "certificates.json",
    "type": "array"
  },
  "skills": {
    "id": "skills",
    "i18nKey": "module.skills",
    "required": true,
    "file": "skills.json",
    "type": "object"
  }
}
```

**配置Schema验证** (schemas/modules.schema.json):
```json
{
  "$schema": "http://json-schema.org/draft-07/schema#",
  "title": "模块配置 Schema",
  "description": "数据模块配置的验证规则",
  "type": "object",
  "required": ["_meta"],
  "properties": {
    "_meta": {
      "type": "object",
      "required": ["version", "lastModified", "schema", "description"],
      "properties": {
        "version": { "type": "string", "pattern": "^\\d+\\.\\d+\\.\\d+$" },
        "lastModified": { "type": "string", "format": "date-time" },
        "schema": { "type": "string", "enum": ["modules"] },
        "description": { "type": "string" }
      }
    }
  },
  "additionalProperties": {
    "type": "object",
    "required": ["id", "i18nKey", "required", "file", "type"],
    "properties": {
      "id": {
        "type": "string",
        "pattern": "^[a-z-]+$",
        "description": "模块ID，只能包含小写字母和连字符"
      },
      "i18nKey": {
        "type": "string",
        "pattern": "^module\\.[a-z-]+$",
        "description": "i18n键值，必须以module.开头"
      },
      "required": {
        "type": "boolean",
        "description": "是否必填"
      },
      "file": {
        "type": "string",
        "pattern": "^[a-z-]+\\.json$",
        "description": "数据文件名"
      },
      "type": {
        "type": "string",
        "enum": ["object", "array"],
        "description": "数据类型"
      }
    }
  }
}
```

**对应的 i18n 配置文件** (languages/[lang]/config-i18n.json):
```json
{
  "module": {
    "personal": "个人信息",
```

**优势**：
- ✅ 添加新模块只需编辑配置文件
- ✅ 定义模块类型（object/array）
- ✅ 控制模块是否必填
- ✅ 前端根据配置自动生成表单

```javascript
// 获取带 i18n 的模块配置
async getModulesConfigWithI18n(langCode) {
  const config = await this.getModulesConfig();
  const i18nConfig = await this.loadConfigI18n(langCode);

  // 合并 i18n 文本
  return Object.keys(config).map(moduleId => ({
    ...config[moduleId],
    name: i18nConfig.module[moduleId] || config[moduleId].i18nKey
  }));
}

// 加载配置 i18n 文件
async loadConfigI18n(langCode) {
  const i18nPath = path.join(__dirname, `../../languages/${langCode}/config-i18n.json`);
  try {
    const data = await fs.readFile(i18nPath, 'utf8');
    return JSON.parse(data);
  } catch (error) {
    // 如果找不到对应语言的 i18n 文件，返回默认（中文）
    const defaultPath = path.join(__dirname, '../../languages/zh-CN/config-i18n.json');
    const data = await fs.readFile(defaultPath, 'utf8');
    return JSON.parse(data);
  }
}

// 获取带 i18n 的技能分类
async getSkillCategoriesWithI18n(langCode) {
  const config = await this.getSkillsConfig();
  const i18nConfig = await this.loadConfigI18n(langCode);

  return config.categories.map(category => ({
    ...category,
    name: i18nConfig.skills[category.id] || category.i18nKey
  }));
}

// 获取带 i18n 的联系字段
async getContactFieldsWithI18n(langCode) {
  const config = await this.getContactFieldsConfig();
  const i18nConfig = await this.loadConfigI18n(langCode);

  return config.contactFields.map(field => {
    const fieldI18n = i18nConfig.contact[field.id] || {};
    return {
      ...field,
      name: fieldI18n.name || field.i18nKey,
      placeholder: fieldI18n.placeholder || ''
    };
  });
}

// 获取带 i18n 的列表配置
async getListConfigWithI18n(listType, langCode) {
  const config = await this.getListConfig(listType);
  const i18nConfig = await this.loadConfigI18n(langCode);

  const listI18n = i18nConfig.list[listType] || {};

  return {
    ...config,
    name: listI18n.name || config.i18nKey,
    fields: config.fields.map(field => ({
      ...field,
      name: listI18n.field?.[field.id] || field.i18nKey

  };
```

## 数据验证

### 配置文件验证机制

#### Schema验证
所有配置文件必须通过JSON Schema验证：
- **模块配置**：`schemas/modules.schema.json`
- **技能分类配置**：`schemas/skills.schema.json`
- **联系字段配置**：`schemas/contact-fields.schema.json`
- **列表类型配置**：`schemas/lists.schema.json`
- **i18n配置**：`schemas/config-i18n.schema.json`

#### 验证服务
```javascript
// services/configValidator.js
const Ajv = require('ajv');
const fs = require('fs').promises;

class ConfigValidator {
  constructor() {
    this.ajv = new Ajv({ allErrors: true });
    this.schemas = {};
  }

  // 加载所有 Schema
  async loadSchemas() {
    const schemaDir = path.join(__dirname, '../schemas');
    const schemaFiles = await fs.readdir(schemaDir);

    for (const file of schemaFiles) {
      if (file.endsWith('.schema.json')) {
        const schemaPath = path.join(schemaDir, file);
        const schemaContent = await fs.readFile(schemaPath, 'utf8');
        const schema = JSON.parse(schemaContent);
        const schemaName = file.replace('.schema.json', '');
        this.schemas[schemaName] = this.ajv.compile(schema);
      }
    }
  }

  // 验证配置文件
  async validateConfig(configPath, schemaName) {
    const schema = this.schemas[schemaName];
    if (!schema) {
      throw new Error(`Schema not found: ${schemaName}`);
    }

    const configContent = await fs.readFile(configPath, 'utf8');
    const config = JSON.parse(configContent);

    const valid = schema(config);

    if (!valid) {
      const errors = schema.errors.map(err => ({
        path: err.instancePath,
        message: err.message,
        params: err.params
      }));

      throw new ConfigValidationError(
        `配置验证失败: ${configPath}`,
        errors
      );
    }

    return true;
  }
}

class ConfigValidationError extends Error {
  constructor(message, errors) {
    super(message);
    this.name = 'ConfigValidationError';
    this.errors = errors;
  }
}
```
